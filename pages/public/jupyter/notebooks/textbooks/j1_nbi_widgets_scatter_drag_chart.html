import nbinteract as nbi
import numpy as np
from numpy import arange

x_points = np.arange(10)
y_points = np.arange(10) + np.random.rand(10)


options = {
    'title': 'Interactive Scatter Plot',
    'animation_duration': 1000,
    'xlabel': 'x-values',
    'ylabel': 'y-values',
    'xlim': (0, 10),
    'ylim': (0, 10)
}

layouts = {
    'plot_height': '480px',
    'plot_width': '800px',
}

show_eqn = False

scatter_drag = nbi.scatter_drag(
    x_points,
    y_points,
    options=options,
    layouts=layouts
)
